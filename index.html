using UnityEngine; using UnityEngine.UI; public class GalaxySkyShooter : MonoBehaviour { public GameObject player; public GameObject bulletPrefab; public GameObject enemyPrefab; public Text scoreText; public Text levelText; int score = 0; int level = 1; float enemySpawnRate = 1.5f; void Start() { InvokeRepeating("SpawnEnemy", 1f, enemySpawnRate); UpdateUI(); } void Update() { MovePlayer(); Shoot(); } void MovePlayer() { float h = Input.GetAxis("Horizontal"); float v = Input.GetAxis("Vertical"); player.transform.Translate(new Vector3(h, v, 0) * 6f * Time.deltaTime); Vector3 pos = Camera.main.WorldToViewportPoint(player.transform.position); pos.x = Mathf.Clamp01(pos.x); pos.y = Mathf.Clamp01(pos.y); player.transform.position = Camera.main.ViewportToWorldPoint(pos); } void Shoot() { if (Input.GetKeyDown(KeyCode.Space)) { Instantiate(bulletPrefab, player.transform.position + Vector3.up * 0.6f, Quaternion.identity); } } void SpawnEnemy() { Vector3 pos = new Vector3(Random.Range(-2.2f, 2.2f), 6f, 0); GameObject enemy = Instantiate(enemyPrefab, pos, Quaternion.identity); enemy.AddComponent(); } public void AddScore() { score += 10; if (score % 100 == 0) { level++; enemySpawnRate = Mathf.Max(0.5f, enemySpawnRate - 0.2f); CancelInvoke(); InvokeRepeating("SpawnEnemy", 1f, enemySpawnRate); } UpdateUI(); } void UpdateUI() { scoreText.text = "Coins: " + score; levelText.text = "Level: " + level; } } public class Bullet : MonoBehaviour { void Update() { transform.Translate(Vector3.up * 10f * Time.deltaTime); Destroy(gameObject, 3f); } void OnTriggerEnter2D(Collider2D col) { if (col.CompareTag("Enemy")) { Destroy(col.gameObject); FindObjectOfType().AddScore(); Destroy(gameObject); } } } public class Enemy : MonoBehaviour { void Update() { transform.Translate(Vector3.down * 2.5f * Time.deltaTime); if (transform.position.y < -6) Destroy(gameObject); } void OnTriggerEnter2D(Collider2D col) { if (col.CompareTag("Player")) { Time.timeScale = 0; Debug.Log("GAME OVER"); } } }
